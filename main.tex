\documentclass[10pt]{ctexbeamer} % chinese template
%\documentclass[10pt]{beamer} % english template

% note taking tool 1: \more{this is your notes}
% manual: https://github.com/xiangrongjingujiu/latex-note-plus
\usepackage[color=blue,smaller]{notePlus}  

% note taking tool 2: \add{this is your notes}
% manual: https://github.com/xiangrongjingujiu/latex-add-notes-simple
\usepackage[color=purple]{addnote} 

% note taking tool 3: \note{this is your notes} inside frames:
% \note commands accumulate and append a single note page 
% after the current slide; If you use multiple \note commands on one slide,
% they “accumulate” and are all shown on the same note.

%\setbeameroption{hide notes} % 隐藏笔记
%\setbeameroption{show only notes} % 只输出笔记
\setbeameroption{show notes} % 在每一页的演示文稿的后面，附上这一页上做的笔记

% 笔记会单独输出在一页幻灯片上，为了在一页笔记幻灯片上容纳更多的文字，需要去除以供展示所用的幻灯片上的页眉页脚等装饰，即使用“plain”模式
\setbeamertemplate{note page}[plain]

% smaller note page font, accommodate more text
% 为了在一页笔记幻灯片上容纳更多的文字，需要设置笔记的字体小一些，这样可以容纳更多文字
% 字体从小到大为：\tiny	、\scriptsize、\footnotesize、\small、\normalsize
% 更多字体大小，参考https://www.overleaf.com/learn/latex/Font_sizes,_families,_and_styles
\setbeamerfont{note page}{size=\scriptsize}
% 修正在演示文稿中使用引用的时候出现的故障，不需要修改
\makeatletter
\defbeameroption{show only notes}[]
{
  \beamer@notestrue
  \beamer@notesnormalsfalse
}
\makeatother
